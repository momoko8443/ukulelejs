function Example03Ctrl() {
	this.myName = "ukulelejs";
	this.sayHello = function() {
		alert(this.myName);
	};
	this.sayHelloWithString = function(str) {
		alert(str);
	};
	this.sayHelloWithArgument = function(name) {
		alert("Hi," + name);
	};
}
