example 03:Event binding between html tag and JS function

  • Demo
  • Html
  • JavaScript 
			
<button uku-onclick="myCtrl.sayHello()">
	Click Me
</button>
<button uku-onclick="myCtrl.sayHelloWithArgument(myCtrl.myName)">
	Click Me with argument
</button>
					    		
		
			
function MyController() {
	this.myName = "please input your name";
	this.sayHello = function() {
		alert(this.myName);
	};
	this.sayHelloWithArgument = function(name) {
		alert("Hi," + name);
	};
}
